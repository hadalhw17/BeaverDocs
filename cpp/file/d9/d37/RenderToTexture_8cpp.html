 30//! Tag that indicates camera that renders to window.

 32struct WindowCamera {};
 33//! Tag that indicates the camera that renders to texture.

 35struct TextureCamera {};

 39    Bvr::Handle<Bvr::CWorld> World;

 42//! Do scene set up once.
 43//! Create WindowCamera and TextureCamera, cube mesh to which texture camera is
 44//! rendering with corresponding material
 45void SetUp(RenderToTextureWorld& simWorld, Bvr::Assets<Bvr::CWorld>& worlds,
 46           Bvr::Assets<Bvr::CMesh>& meshes, Bvr::Assets<Bvr::CImage>& images,
 47           Bvr::Assets<Bvr::PBRMaterial>& materials,
 48           Bvr::Window::Windows&          windows,
 49           const Bvr::Window::MainWindow& mainWnd) {
 50
 51    auto& wnd       = *windows.Get(mainWnd);
 52
 53    // Keep the reference on simulation world
 54    simWorld.World  = worlds.Create();
 55    auto& world     = *worlds.GetMut(simWorld.World);
 56
 57    wnd.Name("RenderToTexture");
 58
 59    // Environment
 60    {
 61        const Bvr::TEntity hEnv = world.CreateEntity("Environment");
 62        world.AddComponent<Bvr::Sun>(hEnv).SunIntensity = 1000000.f;
 63        auto& props                            = world.GetWorldSettings();
 64        props.Environment.Fog.scatteringAmount = 0;
 65        props.Shadows.MaxShadowRange           = 0.1f; // Disable shadow
 66        props.PostEffects.HDR.MinEV = props.PostEffects.HDR.MaxEV = 17.f;
 67    }
 68
 69    // Where we are rendering to
 70    auto hRenderTarget =
 71        images.Create(Bvr::CImage::RenderTarget(500, 500, false)
 72                          .SetName("RenderToTextureOutput"));
 73
 74    // Render to window camera
 75    {
 76        const Bvr::TEntity hCam = world.CreateEntity("WindowCamera");
 77        world.AddComponent<WindowCamera>(hCam);
 78        world.AddComponent<Bvr::TransformComponent>(hCam)
 79            .LookAt(MathConstants::BackVec, MathConstants::UpVec)
 80            .m_translation =
 81            MathConstants::FrontVec * 3.f + MathConstants::UpVec;
 82        world.AddComponent<Bvr::Camera>(hCam, Bvr::Camera::Perspective(64))
 83            .IsActive = true;
 84
 85        world.AddComponent<Bvr::SRenderInfo>(hCam,
 86                                               Bvr::SRenderInfo::Default());
 87        world.AddComponent<Bvr::TRenderTarget>(hCam, wnd.GetId());
 88    }
 89
 90    // Render to texture camera
 91    {
 92        const Bvr::TEntity hCam = world.CreateEntity("TextureCamera");
 93        world.AddComponent<TextureCamera>(hCam);
 94        world.AddComponent<Bvr::TransformComponent>(hCam)
 95            .LookAt(MathConstants::BackVec, MathConstants::UpVec)
 96            .m_translation =
 97            MathConstants::FrontVec * 10.f + MathConstants::UpVec;
 98        world.AddComponent<Bvr::Camera>(hCam, Bvr::Camera::Perspective(64))
 99            .IsActive = true;
100
101        world.AddComponent<Bvr::SRenderInfo>(hCam, Bvr::SRenderInfo::Default());
102        world.AddComponent<Bvr::TRenderTarget>(hCam, hRenderTarget);
103    }
104
105    {
106        // Cube mesh
107        const auto hMeshEnt = world.CreateEntity("Cube");
108        world.AddComponent<Bvr::TransformComponent>(hMeshEnt).m_translation =
109            MathConstants::UpVec;
110        world.AddComponent<Bvr::Handle<Bvr::CMesh>>(
111            hMeshEnt, meshes.Create(Bvr::Geometry::Cube(2, 2, 2)));
112        world.AddComponent<Bvr::Handle<Bvr::PBRMaterial>>(
113            hMeshEnt, materials.Create(Bvr::PBRMaterial{
114                          .m_shaderName    = "GBuffer",
115                          .m_isDoubleSided = true,
116                          .m_bGPUDriven    = true,
117                          .BaseColour      = {1.f, 1.f, 1.f, 1.f},
118                          .DiffuseTex      = hRenderTarget,
119                      }));
120    }
121}

123//! System for cameras input.
124//! WASD to move the TextureCamera camera, arrows to move the WindowCamera
125void OnKey(Bvr::Query<Bvr::With<TextureCamera, Bvr::TransformComponent>> texCam,
126           Bvr::Query<Bvr::With<WindowCamera, Bvr::TransformComponent>> windCam,
127           const Bvr::Events<Bvr::Event::System::SKeyPress>&        events,
128           const Bvr::CTimestep&                                      ts) {
129    const float dt = ts.GetSeconds() * 2.f;
130    {
131        auto [xform] = texCam.GetSingle();
132
133        for (const auto& e : events) {
134            if (e.ePressedKey == Bvr::EKey::W) {
135                xform.m_translation += xform.Front() * dt;
136            } else if (e.ePressedKey == Bvr::EKey::S) {
137                xform.m_translation -= xform.Front() * dt;
138            } else if (e.ePressedKey == Bvr::EKey::A) {
139                xform.m_translation -= xform.Right() * dt;
140            } else if (e.ePressedKey == Bvr::EKey::D) {
141                xform.m_translation += xform.Right() * dt;
142            }
143        }
144    }
145
146    {
147        auto [xform] = windCam.GetSingle();
148
149        for (const auto& e : events) {
150            if (e.ePressedKey == Bvr::EKey::Up) {
151                xform.m_translation += xform.Front() * dt;
152            } else if (e.ePressedKey == Bvr::EKey::Down) {
153                xform.m_translation -= xform.Front() * dt;
154            } else if (e.ePressedKey == Bvr::EKey::Left) {
155                xform.m_translation -= xform.Right() * dt;
156            } else if (e.ePressedKey == Bvr::EKey::Right) {
157                xform.m_translation += xform.Right() * dt;
158            }
159        }
160    }
161}

163//! Set up and invoke the sample.
164int main(int argc, char* argv[]) {
165
166    InitialiseEngine(Bvr::SEngineStartupParams{
167        .m_pArgv             = argv,
168        .m_argc              = argc,
169        .m_windowWidth       = -1,
170        .m_windowHeight      = -1,
171        .m_bInitialiseEditor = false,
172    });
173
174    Bvr::App::Get()
175        .CreateResource<RenderToTextureWorld>()
176        .AddSystem(Bvr::DefaultStages::PRE_UPDATE, &OnKey)
177        // Run the simulation only if editor is disabled
178        .AddStartupSystem(
179            Bvr::SystemDesc(&SetUp).WithCondition([](Bvr::App& app) {
180                return app.GetResource<IBvrEditor>() ? Bvr::EShouldRun::No
181                                                     : Bvr::EShouldRun::Yes;
182            }))
183        .RunStartupSchedule()
184        .Run();
185
186    return 0;
187}
